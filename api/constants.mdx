## Overview

The constants module exports the deployed contract address and ABI (Application Binary Interface) used throughout the application.

### Import

```typescript
import { ContractAddress, ABI } from "@/constants";
```

---

## ContractAddress

The deployed MemberVote contract address on the target network.

<ParamField body="ContractAddress" type="string">
  Ethereum address of the deployed voting contract
</ParamField>

### Value


### Usage Example

```typescript
import { useReadContract } from "wagmi";
import { ContractAddress, ABI } from "@/constants";

const { data: votes } = useReadContract({
  address: ContractAddress,
  abi: ABI,
  functionName: "getOptionAVotes",
```

<Warning>
  Ensure this address matches your deployed contract on the correct network. Update this value when deploying to mainnet or different testnets.
</Warning>

### Contract Functions

#### Read Functions (View)

<Tabs>
  <Tab title="getWorkflowStation">
    Returns the current voting phase
    
    ```typescript
    functionName: "getWorkflowStation"
    inputs: []
    outputs: [{ type: "uint8" }] // 0, 1, or 2
    ```
    
    **Returns:**
    - `0` - REGISTERING
    - `1` - VOTING
    - `2` - RESETTING
  </Tab>
  
  <Tab title="Vote Counts">
    Get vote tallies for each option
    
    ```typescript
    // Option A
    functionName: "getOptionAVotes"
    outputs: [{ type: "uint256" }]
    
    // Option B
    functionName: "getOptionBVotes"
    outputs: [{ type: "uint256" }]
    ```
  </Tab>
  
  <Tab title="Election Info">
    ```typescript
    // Current election ID
    functionName: "getElectionId"
    outputs: [{ type: "uint256" }]
    
    // Contract owner
    functionName: "getOwner"
    outputs: [{ type: "address" }]
    
    // All voters
    functionName: "getVoters"
    outputs: [{ type: "address[]" }]
    ```
  </Tab>
  
  <Tab title="addressVoted">
    Check which election a voter participated in
    
    ```typescript
    functionName: "addressVoted"
    inputs: [{ name: "voter", type: "address" }]
    outputs: [{ type: "uint256" }] // Election ID
    ```
  </Tab>
</Tabs>

#### Write Functions (State-Changing)

<Tabs>
  <Tab title="vote">
    Cast a vote for an option (requires payment)
    
    ```typescript
    functionName: "vote"
    inputs: [{ name: "option", type: "uint256" }]
    stateMutability: "payable"
    ```
    
    **Usage:**
    ```typescript

      address: ContractAddress,
      abi: ABI,
      functionName: "vote",
    ```

  <Tab title="startVote">

### Full Usage Example

```typescript src/hooks/useMemberVote.ts
import { useReadContract, useWriteContract } from "wagmi";
import { ContractAddress, ABI } from "@/constants";
import { parseEther } from "viem";

export const useMemberVote = () => {
  const { writeContract } = useWriteContract();

  // Read contract state
  const { data: workflowStation } = useReadContract({
    address: ContractAddress,
    abi: ABI,
    functionName: "getWorkflowStation",
  });

  const { data: optionAVotes } = useReadContract({
    address: ContractAddress,
    abi: ABI,
    functionName: "getOptionAVotes",
  });

  // Write to contract
  const castVote = (option: number) => {
    writeContract({
      address: ContractAddress,
      abi: ABI,
      functionName: "vote",
      args: [BigInt(option)],
      value: parseEther("0.01"),
    });
  };

  return {
    workflowStation: workflowStation !== undefined ? Number(workflowStation) : 0,
    optionAVotes: optionAVotes ? Number(optionAVotes) : 0,
    castVote,
  };
};
```
